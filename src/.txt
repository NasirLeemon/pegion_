const line2 = "LOGISTICS AND SUPPLY";
  
const sentance = {
  hidden : { opacity : 1},
  visible  : {
    opacity : 1,
    transition : {
      delay : 0.5,
      staggerChildren : 0.1
    },
  },
}

const letter = {
  hidden : {
    opacity : 0, y : 50,
  },
  visible: {
    opacity : 1,
    y : 0
  },
}

{line2.split('').map((char, index) => {
  return (
    <motion.span key={char + '-' + index} variants={letter}>
      {char}
    </motion.span>
  )
})}

variants={sentance}
initial='hidden'
animate='visible'
